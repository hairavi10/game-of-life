pipeline { 
    agent{ label "MASTER" }
    triggers{ pollSCM('* * * * *') }
    stages{
        stage ('Code Download From SCM'){
            steps{
                git 'https://github.com/hairavi10/game-of-life.git'
            }
        }
        stage('Build Package'){
            steps{
                sh 'mvn package'
                stash name: 'gameoflife.war', includes: 'gameoflife-web/target/gameoflife.war'
            }            
        }
        stage('Deploy'){
            agent{ label "ANSIBLE" }
            steps{
                 dir('/home/ansible'){
                      stash name: 'gameoflife.war', includes: 'gameoflife-web/target/gameoflife.war'                                
                 }
            }
        }   
    }
    post{
        always{
            archiveArtifacts artifacts: 'gameoflife-web/target/*.war', onlyIfSuccessful: true
        }
        
    }

}
